function(username){
	var results = []					//宣告results為空陣列
	query = {"profile.name":username}	//Step 1. 設定查詢式
	cursor = db.students.find(query)	//Step 2. 針對students集合
	while(cursor.hasNext())				//判斷cursor集合指標是否還有資料
	{
		var document = cursor.next()	//從cursor集合內，取出一筆資料
		results.push(document)			//將資料放入results陣列最後
	}
	return results						//回傳results陣列
}
